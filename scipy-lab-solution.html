Use scipy to find the best fit line for the given xs and ys

In [10]:
from scipy import optimize

def fun(x, p):
    m, b = p
    return m * x + b

def residuals(p, y, x):
    return y - fun(x, p)

p0 = [0,0]

plsq = optimize.leastsq(residuals, p0, args=(dat['ys'], dat['xs']))
plsq

def peval(x, p):
    A, k, theta = p
    return A * np.sin(2 * np.pi * k * x + theta)
    
def residuals(p, y, x):
    return y - peval(x, p)

p_initial = [1,1,1]
